var OAUTH_CLIENT_ID = "598357137191-04lp6pqg2ehjhagg87kc5ggeaub2erpc.apps.googleusercontent.com";
var BOOTSTRAP_ADMIN_EMAIL = "jakobburgess321@gmail.com";

// ---- Storage keys ----
var P_ADMINS = "admins_csv";   // CSV emails
var P_VIEWERS = "viewers_csv"; // CSV emails
var P_CYCLE_START = "timesheet_cycle_start"; // YYYY-MM-DD

function _normEmail(email) {
  return String(email || "").trim().toLowerCase();
}

function _csvToSet(csv) {
  var set = {};
  String(csv || "")
    .split(",")
    .map(function (s) { return _normEmail(s); })
    .filter(Boolean)
    .forEach(function (e) { set[e] = true; });
  return set;
}

function _setToCsv(set) {
  return Object.keys(set).sort().join(",");
}

function _props() {
  return PropertiesService.getScriptProperties();
}

function _ensureBootstrapAdmin_() {
  var p = _props();
  var admins = _csvToSet(p.getProperty(P_ADMINS));
  var viewers = _csvToSet(p.getProperty(P_VIEWERS));

  var admin = _normEmail(BOOTSTRAP_ADMIN_EMAIL);
  if (admin) {
    admins[admin] = true;
    viewers[admin] = true; 
  }

  p.setProperty(P_ADMINS, _setToCsv(admins));
  p.setProperty(P_VIEWERS, _setToCsv(viewers));
}

function _json_(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}

/**
 * Verify Google ID token
 */
function _verifyIdToken_(idToken) {
  var token = String(idToken || "").trim();
  if (!token) return { ok: false, error: "Missing token" };

  var url = "https://oauth2.googleapis.com/tokeninfo?id_token=" + encodeURIComponent(token);
  try {
    var resp = UrlFetchApp.fetch(url, { muteHttpExceptions: true });
    if (resp.getResponseCode() !== 200) return { ok: false, error: "Token invalid", details: resp.getContentText() };

    var info = JSON.parse(resp.getContentText());
    if (String(info.aud || "") !== String(OAUTH_CLIENT_ID)) return { ok: false, error: "Audience mismatch" };
    
    var email = _normEmail(info.email);
    if (!email) return { ok: false, error: "No email in token" };

    return { ok: true, email: email };
  } catch (e) {
    return { ok: false, error: "Auth failed: " + e };
  }
}

function _getRoles_(email) {
  _ensureBootstrapAdmin_();
  var p = _props();
  var admins = _csvToSet(p.getProperty(P_ADMINS));
  var viewers = _csvToSet(p.getProperty(P_VIEWERS));
  var em = _normEmail(email);
  return { email: em, isAdmin: !!admins[em], isViewer: !!viewers[em], allowed: !!admins[em] || !!viewers[em] };
}

function _requireAuth_(token) {
  var v = _verifyIdToken_(token);
  if (!v.ok) return v;
  var roles = _getRoles_(v.email);
  if (!roles.allowed) return { ok: false, error: "Access denied for " + roles.email };
  return { ok: true, email: roles.email, isAdmin: roles.isAdmin };
}

function _sheet_() { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Logs") || SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; }

function _toYMD_(d) {
  if (!d) return "";
  try { return Utilities.formatDate(new Date(d), Session.getScriptTimeZone(), "yyyy-MM-dd"); } catch(e) { return String(d); }
}
function _parseYMD_(s) {
  var p = String(s).split("-");
  return new Date(p[0], p[1]-1, p[2]);
}

function doGet(e) {
  var token = e && e.parameter ? e.parameter.token : "";
  var auth = _requireAuth_(token);
  if (!auth.ok) return _json_(auth);

  if (e.parameter.action === "me") return _json_({ ok: true, email: auth.email, role: auth.isAdmin ? "admin" : "viewer" });

  var logsRaw = _sheet_().getDataRange().getValues().slice(1);
  var logs = logsRaw.map(function(r) {
    return { date: _toYMD_(r[0]), relationalType: r[1], attendance: r[2], target: r[3], kpi: r[4], hours: r[5], impact: r[6], notes: r[7], id: r[8], timestamp: r[9] ? Utilities.formatDate(new Date(r[9]), Session.getScriptTimeZone(), "dd MMM HH:mm") : "", engagementTime: r[10] || "" };
  }).filter(function(l) { return l.id; });

  var calRaw = _getCalendarSheet_().getDataRange().getValues().slice(1);
  var projected = calRaw.map(function(r) {
    return { date: _toYMD_(r[0]), title: r[1], hours: r[2], id: r[3], seriesId: r[4] };
  });

  var tsRaw = _getTimesheetsSheet_().getDataRange().getValues().slice(1);
  var submitted = tsRaw.filter(function(r) { return _normEmail(r[4]) === _normEmail(auth.email); }).map(function(r) {
    return { start: _toYMD_(r[1]), end: _toYMD_(r[2]), hours: r[3] };
  });

  var getOpts = function(n) {
    var v = _getConfigSheet_(n).getDataRange().getValues();
    return v.length <= 1 ? [] : v.slice(1).map(function(r) { return String(r[0]).trim(); }).filter(Boolean);
  };

  return _json_({
    ok: true,
    logs: logs,
    projected: projected,
    submitted: submitted,
    options: { relationalTypes: getOpts("RelTypes"), kpis: getOpts("KPIs") },
    cycleStart: _props().getProperty(P_CYCLE_START) || "2025-02-01"
  });
}

function doPost(e) {
  var lock = LockService.getScriptLock();
  try {
    lock.waitLock(10000);
    var payload = JSON.parse(e.postData.contents);
    
    // Diagnostic logic MUST be accessible by admin
    if (payload.action === "test_email" || payload.action === "test_reminder") {
        var authDiag = _requireAuth_(payload.token);
        if (!authDiag.ok || !authDiag.isAdmin) return _json_({ ok: false, error: "Admin only" });
        if (payload.action === "test_email") {
            GmailApp.sendEmail(BOOTSTRAP_ADMIN_EMAIL, "📫 Impact Tracker: Connection Test", "Success! Your tracker is correctly configured.\n\nDate: " + new Date().toLocaleString());
            return _json_({ ok: true, message: "Test email sent to " + BOOTSTRAP_ADMIN_EMAIL });
        }
        if (payload.action === "test_reminder") {
            sendTimesheetReminders(true);
            return _json_({ ok: true, message: "Reminder scan executed." });
        }
    }

    var auth = _requireAuth_(payload.token);
    if (!auth.ok) return _json_(auth);
    var action = payload.action;
    if (action === "me") return _json_({ ok: true, email: auth.email, role: auth.isAdmin ? "admin" : "viewer" });

    if (action === "add") {
      _sheet_().appendRow([_parseYMD_(payload.data.date), payload.data.relationalType, payload.data.attendance, payload.data.target, payload.data.kpi, payload.data.hours, payload.data.impact, payload.data.notes, payload.data.id || "ID-"+Utilities.getUuid(), new Date(), payload.data.engagementTime || ""]);
      return _json_({ ok: true });
    }

    if (action === "edit") {
      var s = _sheet_(); var v = s.getDataRange().getValues();
      for (var i=1; i<v.length; i++) {
        if (String(v[i][8]) === String(payload.newEntry.id)) {
          s.getRange(i+1, 1, 1, 11).setValues([[_parseYMD_(payload.newEntry.date), payload.newEntry.relationalType, payload.newEntry.attendance, payload.newEntry.target, payload.newEntry.kpi, payload.newEntry.hours, payload.newEntry.impact, payload.newEntry.notes, payload.newEntry.id, v[i][9] || new Date(), payload.newEntry.engagementTime || ""]]);
          return _json_({ ok: true });
        }
      }
    }

    if (action === "delete") {
      var s2 = _sheet_(); var v2 = s2.getDataRange().getValues();
      for (var j=1; j<v2.length; j++) {
        if (String(v2[j][8]) === String(payload.id)) { s2.deleteRow(j+1); return _json_({ ok: true }); }
      }
    }

    if (action === "calendar_add") {
      var c = _getCalendarSheet_(); var base = _parseYMD_(payload.data.date); var rep = parseInt(payload.data.repeat || 1);
      var sid = "SERIES-"+Utilities.getUuid().slice(0,8);
      for (var k=0; k<rep; k++) {
        var d = new Date(base.getTime()); d.setDate(d.getDate() + (k*7));
        c.appendRow([d, payload.data.title, payload.data.hours, "CAL-"+Utilities.getUuid(), sid]);
      }
      return _json_({ ok: true });
    }

    if (action === "calendar_delete") {
      var c2 = _getCalendarSheet_(); var v3 = c2.getDataRange().getValues();
      for (var l=1; l<v3.length; l++) { if (String(v3[l][3]) === String(payload.id)) { c2.deleteRow(l+1); return _json_({ ok: true }); } }
    }

    if (action === "calendar_move") {
      var c3 = _getCalendarSheet_(); var v4 = c3.getDataRange().getValues();
      for (var m=1; m<v4.length; m++) {
        if (String(v4[m][3]) === String(payload.calendarId)) {
          var r = v4[m]; c3.deleteRow(m+1);
          _sheet_().appendRow([r[0], "Meeting", 1, "Sourced from Calendar", "Outcome Planned", r[2], 0, r[1], "ID-"+Utilities.getUuid(), new Date(), ""]);
          return _json_({ ok: true });
        }
      }
    }

    if (action === "timesheet_submit") {
      var body = _generateTimesheetHtml_(payload.periodStart, payload.periodEnd, payload.summary, payload.totalHours);
      GmailApp.sendEmail(auth.email, "Timesheet Submission: " + payload.periodStart + " to " + payload.periodEnd, "", { htmlBody: body });
      _getTimesheetsSheet_().appendRow([new Date(), _parseYMD_(payload.periodStart), _parseYMD_(payload.periodEnd), payload.totalHours, auth.email]);
      return _json_({ ok: true });
    }

    if (action === "options_set") {
      if (!auth.isAdmin) return _json_({ ok: false, error: "Admin only" });
      var write = function(n, l) {
        var sh = _getConfigSheet_(n); sh.clearContents(); sh.getRange(1,1).setValue("Option List").setFontWeight("bold");
        if (l.length) sh.getRange(2,1,l.length,1).setValues(l.map(function(item){ return [item]; }));
      };
      write("RelTypes", payload.options.relationalTypes || []);
      write("KPIs", payload.options.kpis || []);
      return _json_({ ok: true });
    }
    
    if (action === "cycle_start_set") {
      if (!auth.isAdmin) return _json_({ ok: false, error: "Admin only" });
      _props().setProperty(P_CYCLE_START, payload.date);
      return _json_({ ok: true });
    }

    if (action === "admin_list") {
      if (!auth.isAdmin) return _json_({ ok: false, error: "Admin only" });
      var pr = _props();
      return _json_({ ok: true, admins: Object.keys(_csvToSet(pr.getProperty(P_ADMINS))), viewers: Object.keys(_csvToSet(pr.getProperty(P_VIEWERS))) });
    }

    if (action.indexOf("admin_add") === 0 || action.indexOf("admin_remove") === 0) {
        if (!auth.isAdmin) return _json_({ ok: false, error: "Admin only" });
        var targetType = action.indexOf("admin_add_admin") === 0 || action.indexOf("admin_remove_admin") === 0 ? P_ADMINS : P_VIEWERS;
        var set = _csvToSet(_props().getProperty(targetType));
        var emailStr = _normEmail(payload.email);
        if (action.indexOf("add") !== -1) { 
            set[emailStr] = true; 
            if (targetType === P_ADMINS) { // Admins are always viewers
                var vSet = _csvToSet(_props().getProperty(P_VIEWERS));
                vSet[emailStr] = true;
                _props().setProperty(P_VIEWERS, _setToCsv(vSet));
            }
        } else { 
            if (targetType === P_ADMINS && emailStr === _normEmail(BOOTSTRAP_ADMIN_EMAIL)) return _json_({ ok: false, error: "Locked" });
            delete set[emailStr]; 
        }
        _props().setProperty(targetType, _setToCsv(set));
        return _json_({ ok: true });
    }

    return _json_({ ok: false, error: "Unknown action: " + action });
  } catch (e) {
    return _json_({ ok: false, error: String(e) });
  } finally {
    lock.releaseLock();
  }
}

function _generateTimesheetHtml_(pStart, pEnd, pLogs, total) {
  var start = _parseYMD_(pStart);
  function getWeekTable(wn) {
    var rows = ""; var wt = 0; var days = ["Mon","Tue","Wed","Thu","Fri"];
    for (var i=0; i<5; i++) {
      var d = new Date(start); d.setDate(d.getDate() + (wn-1)*7 + i);
      var ymd = _toYMD_(d); var h = parseFloat(pLogs[ymd] || 0); wt += h;
      rows += '<tr><td style="border:1px solid #ddd;padding:8px;">'+days[i]+' ('+ymd+')</td><td style="border:1px solid #ddd;padding:8px;"></td><td style="border:1px solid #ddd;padding:8px;"></td><td style="border:1px solid #ddd;padding:8px;text-align:center;">'+(h?h.toFixed(1):"")+'</td><td style="border:1px solid #ddd;padding:8px;font-size:11px;">'+(h?"Youth Work":"")+'</td></tr>';
    }
    var sh = 0; var sa = new Date(start); sa.setDate(sa.getDate() + (wn-1)*7 + 5); var su = new Date(start); su.setDate(su.getDate() + (wn-1)*7 + 6);
    sh += parseFloat(pLogs[_toYMD_(sa)] || 0); sh += parseFloat(pLogs[_toYMD_(su)] || 0); wt += sh;
    rows += '<tr><td style="border:1px solid #ddd;padding:8px;">Sat/Sun</td><td style="border:1px solid #ddd;padding:8px;"></td><td style="border:1px solid #ddd;padding:8px;"></td><td style="border:1px solid #ddd;padding:8px;text-align:center;">'+(sh?sh.toFixed(1):"")+'</td><td style="border:1px solid #ddd;padding:8px;font-size:11px;">'+(sh?"Youth Work":"")+'</td></tr>';
    return '<table style="width:100%;border-collapse:collapse;margin-top:15px;font-size:13px;"><thead style="background:#f9f9f9;"><tr><th style="border:1px solid #ddd;padding:8px;text-align:left;">Day</th><th style="border:1px solid #ddd;padding:8px;">Start</th><th style="border:1px solid #ddd;padding:8px;">End</th><th style="border:1px solid #ddd;padding:8px;">Hrs</th><th style="border:1px solid #ddd;padding:8px;">Notes</th></tr></thead><tbody>'+rows+'<tr style="font-weight:bold;background:#fafafa;"><td colspan="3" style="border:1px solid #ddd;padding:8px;text-align:right;">Week Total</td><td style="border:1px solid #ddd;padding:8px;text-align:center;">'+wt.toFixed(1)+'</td><td style="border:1px solid #ddd;padding:8px;"></td></tr></tbody></table>';
  }
  var head = '<div style="margin-bottom:20px;"><h2 style="margin:0;">Employment Timesheet</h2><p style="margin:5px 0;color:#666;">Period: '+pStart+' to '+pEnd+'</p></div>';
  return '<div style="font-family:sans-serif;max-width:700px;margin:auto;color:#333;padding:20px;border:1px solid #eee;">'+head+getWeekTable(1)+getWeekTable(2)+'<div style="margin-top:25px;padding:15px;background:#f0f7ff;text-align:right;border-radius:8px;"><span style="font-size:12px;text-transform:uppercase;color:#5583b4;">Fortnight Total</span><div style="font-size:24px;font-weight:bold;color:#1e40af;">'+parseFloat(total).toFixed(1)+'h</div></div></div>';
}

function sendTimesheetReminders(force) {
  var p = _props(); var start = p.getProperty(P_CYCLE_START) || "2025-02-01";
  var ref = _parseYMD_(start); var now = new Date(); var today = _toYMD_(now);
  var curr = new Date(ref); var deadline = null;
  for (var i=0; i<100; i++) {
    var pEnd = new Date(curr); pEnd.setDate(pEnd.getDate() + 13);
    while (pEnd.getDay() !== 0) pEnd.setDate(pEnd.getDate() + 1);
    var fri = new Date(pEnd); fri.setDate(fri.getDate() - 2);
    if (_toYMD_(fri) === today || force === true) { deadline = _toYMD_(pEnd); break; }
    if (pEnd > now && _toYMD_(pEnd) !== today) break;
    curr = new Date(pEnd); curr.setDate(curr.getDate() + 1);
  }
  if (deadline) {
    GmailApp.sendEmail(BOOTSTRAP_ADMIN_EMAIL, "🚨 Timesheet Reminder: Period ends Sunday " + deadline, "Hi,\n\nThis is a reminder that your timesheet period ends this Sunday, " + deadline + ". Please update your logs.\n\nThanks!");
  }
}

function setupReminders() {
  var triggers = ScriptApp.getProjectTriggers();
  for (var i=0; i<triggers.length; i++) { if (triggers[i].getHandlerFunction() === "sendTimesheetReminders") ScriptApp.deleteTrigger(triggers[i]); }
  ScriptApp.newTrigger("sendTimesheetReminders").timeBased().onWeekDay(ScriptApp.WeekDay.FRIDAY).atHour(8).create();
  return "Scheduled!";
}

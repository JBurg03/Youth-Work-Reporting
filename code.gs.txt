/**
 * Verify Google ID token with Google's tokeninfo endpoint.
 * IMPORTANT: We avoid custom request headers from the browser to prevent CORS preflight;
 * the token is sent in query/body and verified server-side.
 */
function _verifyIdToken_(idToken) {
  var token = String(idToken || "").trim();
  if (!token) return { ok: false, error: "Missing token" };

  // Cache verification briefly (token is a JWT; we rely on Google tokeninfo)
  var cache = CacheService.getScriptCache();
  var digest = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, token);
  var cacheKey = "jwt:" + digest.map(function(b) {
    var hex = (b < 0 ? b + 256 : b).toString(16);
    return hex.length === 1 ? "0" + hex : hex;
  }).join("").slice(0, 50);

  var cached = cache.get(cacheKey);
  if (cached) {
    try { return JSON.parse(cached); } catch (e) {}
  }

  var url = "https://oauth2.googleapis.com/tokeninfo?id_token=" + encodeURIComponent(token);
  try {
    var resp = UrlFetchApp.fetch(url, { muteHttpExceptions: true, followRedirects: true });
    var code = resp.getResponseCode();
    var body = resp.getContentText();

    if (code !== 200) {
      var outBad = { ok: false, error: "Token invalid", code: "AUTH-01", details: body.slice(0, 300) };
      cache.put(cacheKey, JSON.stringify(outBad), 30);
      return outBad;
    }

    var info = JSON.parse(body);

    // Validate audience and basic fields
    if (String(info.aud || "") !== String(OAUTH_CLIENT_ID)) {
      var outAud = { ok: false, error: "Token audience mismatch", code: "AUTH-02" };
      cache.put(cacheKey, JSON.stringify(outAud), 30);
      return outAud;
    }

    if (String(info.email_verified || "") !== "true") {
      var outVer = { ok: false, error: "Email not verified" };
      cache.put(cacheKey, JSON.stringify(outVer), 30);
      return outVer;
    }

    var email = _normEmail(info.email);
    if (!email) {
      var outEmail = { ok: false, error: "No email in token" };
      cache.put(cacheKey, JSON.stringify(outEmail), 30);
      return outEmail;
    }

    var outOk = {
      ok: true,
      email: email,
      name: info.name || "",
      picture: info.picture || "",
      exp: info.exp || ""
    };

    cache.put(cacheKey, JSON.stringify(outOk), 60);
    return outOk;
  } catch (e) {
    return { ok: false, error: "Token verification failed: " + e };
  }
}

function _getRoles_(email) {
  _ensureBootstrapAdmin_();

  var em = _normEmail(email);
  var usersSheet = _getUsersSheet_();
  var data = usersSheet.getDataRange().getValues();
  
  var isAdmin = false;
  var isViewer = false;
  var isUser = false;
  var alias = em; // default

  for (var i = 1; i < data.length; i++) {
    if (_normEmail(data[i][0]) === em) {
      alias = String(data[i][1] || em).trim();
      var role = String(data[i][2] || "").toLowerCase();
      isAdmin = (role === "admin");
      isViewer = (role === "viewer");
      isUser = true; // anyone on the list is a user (or viewer)
      break;
    }
  }

  return { email: em, alias: alias, isAdmin: isAdmin, isViewer: isViewer, isUser: isUser, allowed: isUser };
}

function _requireAuth_(token) {
  var v = _verifyIdToken_(token);
  if (!v.ok) return { ok: false, error: v.error || "Unauthorized", details: v.details || "" };

  var roles = _getRoles_(v.email);
  if (!roles.allowed) return { ok: false, error: "Access denied for " + roles.email };

  // Return the highest role string
  var roleLabel = "user";
  if (roles.isAdmin) roleLabel = "admin";
  else if (roles.isViewer) roleLabel = "viewer";

  return { 
    ok: true, 
    email: roles.email, 
    alias: roles.alias, 
    isAdmin: roles.isAdmin, 
    isViewer: roles.isViewer, 
    isUser: roles.isUser,
    role: roleLabel
  };
}

/**
 * Standardize YYYY-MM-DD for internal logic/storage.
 */
function _toYMD_(value) {
  if (value === null || value === undefined || value === "") return "";
  if (Object.prototype.toString.call(value) === "[object Date]" && !isNaN(value.getTime())) {
    return Utilities.formatDate(value, Session.getScriptTimeZone(), "yyyy-MM-dd");
  }
  var s = String(value).trim();
  if (/^\d{4}-\d{2}-\d{2}$/.test(s)) return s;
  var d = new Date(s);
  if (!isNaN(d.getTime())) return Utilities.formatDate(d, Session.getScriptTimeZone(), "yyyy-MM-dd");
  return s;
}

/**
 * Global display format: DD/MM/YY
 */
function _toDMY_(value) {
  if (!value) return "";
  try {
    var d = (Object.prototype.toString.call(value) === "[object Date]") ? value : new Date(value);
    if (isNaN(d.getTime())) return String(value);
    return Utilities.formatDate(d, Session.getScriptTimeZone(), "dd/MM/yy");
  } catch (e) { return String(value); }
}

function _formatTimestamp_(value) {
  if (!value) return "";
  try {
    var d = (Object.prototype.toString.call(value) === "[object Date]") ? value : new Date(value);
    if (isNaN(d.getTime())) return String(value);
    return Utilities.formatDate(d, Session.getScriptTimeZone(), "dd/MM/yy HH:mm");
  } catch (e) { return String(value); }
}

function _parseYMD_(ymdStr) {
  var s = String(ymdStr || "").trim();
  if (!s) return "";
  var m = /^(\d{4})-(\d{2})-(\d{2})$/.exec(s);
  if (m) return new Date(Number(m[1]), Number(m[2]) - 1, Number(m[3]));
  var d = new Date(s);
  return isNaN(d.getTime()) ? "" : d;
}

/**
 * Seeds NZ school term dates for the current and previous year.
 */
function _seedTermDefaults_(sheet) {
  var currentYear = new Date().getFullYear();
  var terms = _getNZTermDates_(currentYear);
  var prevTerms = _getNZTermDates_(currentYear - 1);
  var allTerms = prevTerms.concat(terms);
  if (allTerms.length > 0) {
    var rows = allTerms.map(function(t) { return [t.name, t.start, t.end, t.goal, t.year]; });
    sheet.getRange(2, 1, rows.length, 5).setValues(rows);
  }
}

function doGet(e) {
  var token = e && e.parameter ? e.parameter.token : "";
  var auth = _requireAuth_(token);
  if (!auth.ok) return _json_({ ok: false, error: auth.error, details: auth.details || "" });

  // Special GET: auth status
  if (e && e.parameter && e.parameter.action === "me") {
    return _json_({
      ok: true,
      email: auth.email,
      role: auth.role
    });
  }

  var sheet = _sheet_();
  var values = sheet.getDataRange().getValues();

  if (!values || values.length <= 1) {
    return _json_({ ok: true, role: auth.role, logs: [] });
  }

  // Ensure IDs exist
  var lock = LockService.getScriptLock();
  var logs = [];
  try {
    lock.waitLock(30000);

    for (var i = 1; i < values.length; i++) {
      var row = values[i];
      var rowIndex = i + 1;

      var id = row[8];
      if (!id) {
        id = "ID-" + Utilities.getUuid();
        sheet.getRange(rowIndex, 9).setValue(id);
        row[8] = id;
      }

      logs.push({
        date: _toYMD_(row[0]),
        relationalType: row[1],
        attendance: row[2],
        target: row[3],
        kpi: row[4],
        hours: row[5],
        impact: row[6],
        notes: row[7],
        id: String(id),
        timestamp: _formatTimestamp_(row[9]),
        engagementTime: row[10] || "",
        addedBy: row[11] || "",
        lastEditedBy: row[12] || "",
        lastEditedAt: _formatTimestamp_(row[13]) || "",
        yearLevel: row[14] || "",
        referralPathway: row[15] || ""
      });
    }
  } finally {
    lock.releaseLock();
  }

  // Load saved dropdown options from multi-sheet config
  var relS = _getConfigSheet_("RelTypes");
  var kpiS = _getConfigSheet_("KPIs");
  
  var getVals = function(s) {
    var v = s.getDataRange().getValues();
    if (v.length <= 1) return [];
    return v.slice(1).map(function(r) { return String(r[0] || "").trim(); }).filter(Boolean);
  };

  var options = {
    relationalTypes: getVals(relS),
    kpis: getVals(kpiS)
  };

  // -- Fetch Calendar (Projected) --
  var calSheet = _getCalendarSheet_();
  var calValues = calSheet.getDataRange().getValues();
  var projected = [];
  for (var k = 1; k < calValues.length; k++) {
    var cRow = calValues[k];
    projected.push({
      date: _toYMD_(cRow[0]),
      title: cRow[1],
      hours: cRow[2],
      id: String(cRow[3]),
      seriesId: String(cRow[4]),
      users: String(cRow[5] || "")
    });
  }

  // -- Fetch Timesheet Submission History --
  var tsSheet = _getTimesheetsSheet_();
  var tsValues = tsSheet.getDataRange().getValues();
  var submitted = [];
  for (var m = 1; m < tsValues.length; m++) {
    var tsRow = tsValues[m];
    submitted.push({
      start: _toYMD_(tsRow[1]),
      end: _toYMD_(tsRow[2]),
      hours: tsRow[3],
      email: tsRow[4] || "",
      approved: tsRow[5] || false
    });
  }

  // -- Fetch Terms --
  var termsSheet = _getTermsSheet_();
  var termsValues = termsSheet.getDataRange().getValues();
  var terms = [];
  for (var t = 1; t < termsValues.length; t++) {
    var tRow = termsValues[t];
    terms.push({
      name: String(tRow[0] || ""),
      start: _toYMD_(tRow[1]),
      end: _toYMD_(tRow[2]),
      goal: parseFloat(tRow[3]) || 100,
      year: parseInt(tRow[4], 10) || new Date().getFullYear()
    });
  }

  // -- Fetch Users (Registry) --
  var uSheet = _getUsersSheet_();
  var uValues = uSheet.getDataRange().getValues();
  var users = [];
  for (var u = 1; u < uValues.length; u++) {
    users.push({
      email: _normEmail(uValues[u][0]),
      alias: String(uValues[u][1] || "").trim(),
      role: String(uValues[u][2] || "").toLowerCase(),
      reminders: String(uValues[u][3] || "").toUpperCase() === "TRUE"
    });
  }

  return _json_({
    ok: true,
    email: auth.email,
    alias: auth.alias,
    role: auth.role,
    users: users,
    logs: logs,
    projected: projected,
    submitted: submitted,
    options: options,
    terms: terms,
    cycleStart: _props().getProperty(P_CYCLE_START) || "2025-02-01"
  });
}

function doPost(e) {
  var lock = LockService.getScriptLock();
  try {
    lock.waitLock(30000);

    var raw = (e && e.postData && e.postData.contents) ? e.postData.contents : "";
    var payload = {};
    try { payload = JSON.parse(raw || "{}"); } catch (err) {
      return _json_({ ok: false, error: "Invalid JSON" });
    }

    var auth = _requireAuth_(payload.token);
    if (!auth.ok) return _json_({ ok: false, error: auth.error, details: auth.details || "" });

    var action = String(payload.action || "").trim().toLowerCase();
    
    // ---- Global Actions ----
    if (action === "me") return _json_({ ok: true, email: auth.email, role: auth.role });


    // ---- Admin panel actions (no sheet writes, just allowlist) ----
    if (action === "admin_list") {
      if (!auth.isAdmin) return _json_({ ok: false, error: "Admin only" });
      _ensureBootstrapAdmin_();
      
      var uSheet = _getUsersSheet_();
      var uValues = uSheet.getDataRange().getValues();
      var users = [];
      for (var u = 1; u < uValues.length; u++) {
        users.push({
          email: _normEmail(uValues[u][0]),
          alias: String(uValues[u][1] || "").trim(),
          role: String(uValues[u][2] || "").toLowerCase(),
          reminders: String(uValues[u][3] || "").toUpperCase() === "TRUE"
        });
      }
      return _json_({ ok: true, users: users });
    }
    else if (action === "users_set") {
      if (!auth.isAdmin) return _json_({ ok: false, error: "Admin only" });
      var newUsers = payload.users;
      if (!Array.isArray(newUsers)) return _json_({ ok: false, error: "Missing users array" });

      var uSheet = _getUsersSheet_();
      uSheet.clearContents();
      uSheet.getRange(1, 1, 1, 4).setValues([["Email", "Shorthand", "Role", "SendReminders"]]).setFontWeight("bold");
      
      if (newUsers.length > 0) {
        var rows = newUsers.map(function(u) {
          return [
            _normEmail(u.email),
            String(u.alias || u.email).trim(),
            String(u.role || "user").toLowerCase(),
            String(u.reminders === true).toUpperCase()
          ];
        });
        uSheet.getRange(2, 1, rows.length, 4).setValues(rows);
      }
      return _json_({ ok: true });
    }
    else if (action === "test_email") {
      if (!auth.isAdmin) return _json_({ ok: false, error: "Admin only", code: "PERM-01" });
      GmailApp.sendEmail(PERSONAL_ADMIN_EMAIL, "Impact Tracker: Connection Test", "Success! Your tracker is correctly configured to send emails.\n\nSent from: " + NOTIFICATION_EMAIL + "\nDate: " + new Date().toLocaleString());
      return _json_({ ok: true, message: "Test email sent to " + PERSONAL_ADMIN_EMAIL });
    }
    else if (action === "test_reminder") {
      if (!auth.isAdmin) return _json_({ ok: false, error: "Admin only" });
      // Force trigger the reminder logic regardless of day
      sendTimesheetReminders(true); 
      return _json_({ ok: true, message: "Reminder scan executed. Check script logs/email." });
    }

    // ---- Dropdown options actions (admin only) ----
    else if (action === "options_get") {
      if (!auth.isAdmin) return _json_({ ok: false, error: "Admin only" });
      
      var relS_G = _getConfigSheet_("RelTypes");
      var kpiS_G = _getConfigSheet_("KPIs");
      
      var getV = function(s) {
        var v = s.getDataRange().getValues();
        if (v.length <= 1) return [];
        return v.slice(1).map(function(r) { return String(r[0] || "").trim(); }).filter(Boolean);
      };

      return _json_({
        ok: true, 
        options: {
          relationalTypes: getV(relS_G),
          kpis: getV(kpiS_G)
        }
      });
    }
    else if (action === "options_set") {
      if (!auth.isAdmin) return _json_({ ok: false, error: "Admin only" });
      var newOpts = payload.options;
      if (!newOpts || typeof newOpts !== "object") return _json_({ ok: false, error: "Missing options object" });

      var relTypes = (Array.isArray(newOpts.relationalTypes) ? newOpts.relationalTypes : [])
        .map(function (s) { return String(s || "").trim(); }).filter(Boolean);
      var kpis = (Array.isArray(newOpts.kpis) ? newOpts.kpis : [])
        .map(function (s) { return String(s || "").trim(); }).filter(Boolean);

      var writeToSheet = function(sheetName, list) {
        var s = _getConfigSheet_(sheetName);
        s.clearContents();
        s.getRange(1, 1).setValue("Option List").setFontWeight("bold");
        if (list.length > 0) {
          var rows = list.map(function(v) { return [v]; });
          s.getRange(2, 1, rows.length, 1).setValues(rows);
        }
      };

      writeToSheet("RelTypes", relTypes);
      writeToSheet("KPIs", kpis);

      return _json_({ ok: true, options: { relationalTypes: relTypes, kpis: kpis } });
    }

    // ---- Viewer protection ----
    var readOnlyActions = ["options_get", "admin_list", "terms_get", "me"];
    if (auth.isViewer && readOnlyActions.indexOf(action) === -1) {
      return _json_({ ok: false, error: "Access Denied: Viewer role is read-only.", code: "VIEW-01" });
    }

    // ---- Sheet data actions (all authenticated users) ----
    var sheet = _sheet_();
    var values = sheet.getDataRange().getValues();

    if (action === "delete") {
      var deleteId = String(payload.id || "");
      if (!deleteId) return _json_({ ok: false, error: "Missing id" });

      for (var i = 1; i < values.length; i++) {
        if (String(values[i][8]) === deleteId) {

/**
 * AUTOMATED WEEKLY REMINDER
 * Triggered every Friday. Checks if the next Sunday is a timesheet deadline.
 * Includes a summary of hours logged to date in the email.
 */
function sendTimesheetReminders(force) {
  var p = _props();
  var cycleStart = p.getProperty(P_CYCLE_START) || "2025-02-01";
  // Send reminders to your personal email so you can reply "Confirm Hours"
  var adminEmail = PERSONAL_ADMIN_EMAIL;
  
  var startRef = new Date(cycleStart + "T00:00:00");
  var now = new Date();
  var todayStr = _toYMD_(now);
  
  var current = new Date(startRef);
  var deadlineSunday = null;
  var periodStart = null;
  
  for (var i = 0; i < 100; i++) {
    var pStart = new Date(current);
    var pEnd = new Date(current);
    pEnd.setDate(pEnd.getDate() + 13);
    while (pEnd.getDay() !== 0) pEnd.setDate(pEnd.getDate() + 1);
    
    var pEndStr = _toYMD_(pEnd);
    
    var fridayBefore = new Date(pEnd);
    fridayBefore.setDate(fridayBefore.getDate() - 2);
    
    if (_toYMD_(fridayBefore) === todayStr || force === true) {
      deadlineSunday = pEndStr;
      periodStart = _toYMD_(pStart);
      break;
    }
    
    if (pEnd > now && _toYMD_(pEnd) !== todayStr) break;
    
    current = new Date(pEnd);
    current.setDate(current.getDate() + 1);
  }
  
  if (!deadlineSunday || !periodStart) {
    console.log("No deadline detected for this Sunday.");
    return;
  }

  // Fetch users with reminders enabled
  var uSheet = _getUsersSheet_();
  var uData = uSheet.getDataRange().getValues();
  var activeUsers = [];
  for (var u = 1; u < uData.length; u++) {
    if (String(uData[u][3]).toUpperCase() === "TRUE") {
      activeUsers.push({ email: _normEmail(uData[u][0]), alias: String(uData[u][1] || uData[u][0]) });
    }
  }

  if (activeUsers.length === 0) {
    console.log("No users have reminders enabled.");
    return;
  }

  // Fetch all logs once
  var logValues = _sheet_().getDataRange().getValues();
  
  activeUsers.forEach(function(user) {
    var userHours = 0;
    var dailySummary = {};
    for (var r = 1; r < logValues.length; r++) {
      var logDate = _toYMD_(logValues[r][0]);
      var addedBy = _normEmail(logValues[r][11]);
      if (addedBy === user.email && logDate >= periodStart && logDate <= deadlineSunday) {
        var hrs = parseFloat(logValues[r][5]) || 0;
        userHours += hrs;
        dailySummary[logDate] = (dailySummary[logDate] || 0) + hrs;
      }
    }
    
    var summaryHtml = "<table style='width:100%;border-collapse:collapse;margin:15px 0;'>";
    summaryHtml += "<tr style='background:#f5f5f5;'><th style='border:1px solid #ddd;padding:8px;text-align:left;'>Date</th><th style='border:1px solid #ddd;padding:8px;text-align:center;'>Hours</th></tr>";
    var sortedDates = Object.keys(dailySummary).sort();
    for (var d = 0; d < sortedDates.length; d++) {
      summaryHtml += "<tr><td style='border:1px solid #ddd;padding:8px;'>" + _toDMY_(sortedDates[d]) + "</td><td style='border:1px solid #ddd;padding:8px;text-align:center;font-weight:bold;'>" + dailySummary[sortedDates[d]].toFixed(1) + "h</td></tr>";
    }
    summaryHtml += "<tr style='background:#f0f9ff;font-weight:bold;'><td style='border:1px solid #ddd;padding:8px;'>TOTAL (" + user.alias + ")</td><td style='border:1px solid #ddd;padding:8px;text-align:center;'>" + userHours.toFixed(1) + "h</td></tr>";
    summaryHtml += "</table>";
    
    var subject = "Timesheet Reminder: " + userHours.toFixed(1) + "h logged — Period ending " + _toDMY_(deadlineSunday);
    var htmlBody = "<div style='font-family:sans-serif;color:#333;max-width:600px;margin:20px auto;'>" +
      "<h2 style='color:#1e40af;'>Timesheet Reminder</h2>" +
      "<p>Hi " + user.alias + ",</p>" +
      "<p>Your fortnightly timesheet period <strong>" + _toDMY_(periodStart) + "</strong> to <strong>" + _toDMY_(deadlineSunday) + "</strong> ends this Sunday.</p>" +
      "<h3 style='margin-top:20px;'>Your Hours Summary to Date</h3>" +
      summaryHtml +
      "<p style='margin-top:20px;background:#f0fdf4;padding:15px;border-radius:8px;border:1px solid #bbf7d0;'>" +
        "<strong>To submit and approve:</strong> Simply reply to this email with the text <code>Confirm Hours</code> in the first line." +
      "</p>" +
      "<p style='font-size:12px;color:#666;margin-top:20px;'>This is an automated reminder from the Youth Work Tracker.</p>" +
    "</div>";

    try {
      MailApp.sendEmail({ to: user.email, subject: subject, htmlBody: htmlBody });
      console.log("Reminder sent to: " + user.email);
    } catch (e) {
      console.error("Failed to send reminder to " + user.email + ": " + e);
    }
  });
}

/**
 * Scans inbox for replies containing "Confirm Hours" to timesheet reminders.
 * Marks the relevant timesheet period as approved.
 */
function checkTimesheetConfirmations() {
  console.log("Starting timesheet confirmation scan...");
  
  // Get valid users
  var uSheet = _getUsersSheet_();
  var uData = uSheet.getDataRange().getValues();
  var validUsers = {};
  for (var u = 1; u < uData.length; u++) {
    validUsers[_normEmail(uData[u][0])] = String(uData[u][1] || uData[u][0]);
  }

  try {
    var threads = GmailApp.search('subject:"Timesheet Reminder" newer_than:7d', 0, 20);
    for (var t = 0; t < threads.length; t++) {
      var messages = threads[t].getMessages();
      for (var m = 0; m < messages.length; m++) {
        var msg = messages[m];
        var from = _normEmail(msg.getFrom().replace(/.*</, "").replace(/>.*/, ""));
        
        if (!validUsers[from]) continue;
        if (!/confirm\s*hours/i.test(msg.getPlainBody())) continue;
        
        var subject = threads[t].getFirstMessageSubject();
        // Match both ISO (YYYY-MM-DD) and DD/MM/YY
        var periodMatch = /Period ending\s+([\d/-]+)/.exec(subject);
        if (!periodMatch) continue;
        
        var dateStr = periodMatch[1];
        var targetSunday = "";
        if (dateStr.indexOf("/") > -1) {
           var parts = dateStr.split("/");
           targetSunday = "20" + parts[2] + "-" + parts[1] + "-" + parts[0];
        } else {
           targetSunday = dateStr;
        }
        
        _approveTimesheetForUser(from, targetSunday);
        msg.markRead();
        threads[t].moveToArchive();
        break;
      }
    }
  } catch (e) {
    console.error("checkTimesheetConfirmations error: " + e);
  }
}

function _approveTimesheetForUser(userEmail, sundayYMD) {
  var tsSheet = _getTimesheetsSheet_();
  var data = tsSheet.getDataRange().getValues();
  userEmail = _normEmail(userEmail);
  var found = false;

  for (var i = 1; i < data.length; i++) {
    var rowSunday = _toYMD_(data[i][2]);
    var rowEmail = _normEmail(data[i][4]);
    if (rowSunday === sundayYMD && rowEmail === userEmail && !data[i][5]) {
      tsSheet.getRange(i + 1, 6).setValue(true);
      found = true;
      break;
    }
  }
  
  if (!found) {
    // Check if ALREADY approved (double-processing prevention)
    var alreadyApproved = data.some(function(r) { 
      return _toYMD_(r[2]) === sundayYMD && _normEmail(r[4]) === userEmail && r[5] === true; 
    });
    if (alreadyApproved) return;

    // Direct Approval Calculation
    var cycleStart = _props().getProperty(P_CYCLE_START) || "2025-02-01";
    var pStart = "";
    var current = new Date(cycleStart + "T00:00:00");
    for (var j = 0; j < 150; j++) {
      var s = new Date(current);
      var e = new Date(current);
      e.setDate(e.getDate() + 13);
      while (e.getDay() !== 0) e.setDate(e.getDate() + 1);
      if (_toYMD_(e) === sundayYMD) { pStart = _toYMD_(s); break; }
      current = new Date(e); current.setDate(current.getDate() + 1);
    }
    
    if (pStart) {
      var logs = _sheet_().getDataRange().getValues();
      var total = 0;
      for (var l = 1; l < logs.length; l++) {
        var d = _toYMD_(logs[l][0]);
        var u = _normEmail(logs[l][11]);
        if (u === userEmail && d >= pStart && d <= sundayYMD) {
          total += parseFloat(logs[l][5]) || 0;
        }
      }
      tsSheet.appendRow([new Date(), pStart, sundayYMD, total, userEmail, true]);
      console.log("✅ Create direct approved timesheet for " + userEmail + " @ " + sundayYMD);
    }
  }

  // Audit and Receipt
  try {
    var appSheet = _getApprovalsSheet_();
    appSheet.appendRow([new Date(), "Email Response", sundayYMD, userEmail]);
    
    GmailApp.sendEmail(userEmail, 
      "Timesheet Approved: Period ending " + _toDMY_(sundayYMD),
      "Hi,\n\nYour timesheet for the period ending " + _toDMY_(sundayYMD) + " has been successfully submitted and approved via email.\n\nDate: " + new Date().toLocaleString());
  } catch (e) { console.error("Receipt failed for " + userEmail); }
}

/**
 * Run this function ONCE in the Apps Script editor to set up all automation triggers.
 */
function setupReminders() {
  var triggers = ScriptApp.getProjectTriggers();
  for (var i = 0; i < triggers.length; i++) {
    var handler = triggers[i].getHandlerFunction();
    if (handler === "sendTimesheetReminders" || handler === "checkTimesheetConfirmations") {
      ScriptApp.deleteTrigger(triggers[i]);
    }
  }
  
  // Friday morning reminder
  ScriptApp.newTrigger("sendTimesheetReminders")
    .timeBased()
    .onWeekDay(ScriptApp.WeekDay.FRIDAY)
    .atHour(8)
    .create();
  
  // Check for email confirmations every 2 hours
  ScriptApp.newTrigger("checkTimesheetConfirmations")
    .timeBased()
    .everyHours(2)
    .create();
    
  return "Successfully scheduled Friday reminders + confirmation checker!";
}

      report += "  - Contains " + messages.length + " messages.\n";
      
      for (var m = 0; m < messages.length; m++) {
        var msg = messages[m];
        var fromRaw = msg.getFrom();
        var from = _normEmail(fromRaw.replace(/.*</, "").replace(/>.*/, ""));
        var bodySnippet = (msg.getPlainBody() || "").slice(0, 60).replace(/\n/g, " ");
        
        report += "    Msg [" + m + "] from [" + from + "]: \"" + bodySnippet + "...\"\n";
        
        var isFromAdmin = (from === adminEmail);
        var hasConfirmText = /confirm\s*hours/i.test(msg.getPlainBody());
        
        if (isFromAdmin && hasConfirmText) {
          report += "    ✅ MATCH FOUND: Admin replied with 'Confirm Hours'.\n";
        } else {
          if (!isFromAdmin) report += "    (skipping: not from admin email)\n";
          if (!hasConfirmText) report += "    (skipping: 'Confirm Hours' not found in body)\n";
        }
      }
    }
  } catch (e) {
    report += "❌ ERROR: " + e + "\n";
  }
  
  report += "\n--- End of Report ---\n";
  console.log(report); // More visible in the Apps Script editor
  return report;

/**
 * Aggregates statistics for a specific term and sends a professional email report.
 */
function _sendTermSummaryEmail_(recipientEmail, termName, year) {
  try {
    var terms = _getTermsSheet_().getDataRange().getValues();
    var termIdx = -1;
    for (var i = 1; i < terms.length; i++) {
        var tName = String(terms[i][0]);
        var tYear = parseInt(terms[i][4], 10);
        if (tName === termName && tYear === parseInt(year, 10)) {
            termIdx = i;
            break;
        }
    }
    if (termIdx === -1) return { ok: false, error: "Term not found: " + termName };
    
    var tData = terms[termIdx];
    var start = _toYMD_(tData[1]);
    var end = _toYMD_(tData[2]);
    var goal = parseFloat(tData[3]) || 100;
    
    var logs = _sheet_().getDataRange().getValues();
    var totalHours = 0;
    var relTypes = {};
    var kpis = {};
    var highImpactCount = 0;
    var userEmail = _normEmail(recipientEmail);
    
    for (var j = 1; j < logs.length; j++) {
        var logDate = _toYMD_(logs[j][0]);
        var addedByStr = String(logs[j][11] || "");
        var addedByList = addedByStr.split(",").map(function(e) { return _normEmail(e); });
        
        if (addedByList.indexOf(userEmail) !== -1 && logDate >= start && logDate <= end) {
            var hrs = parseFloat(logs[j][5]) || 0;
            totalHours += hrs;
            
            var rt = String(logs[j][1] || "Other");
            relTypes[rt] = (relTypes[rt] || 0) + hrs;
            
            var kpi = String(logs[j][4] || "General");
            kpis[kpi] = (kpis[kpi] || 0) + 1;
            
            if (parseInt(logs[j][6]) >= 4) highImpactCount++;
        }
    }
    
    var progress = Math.min(100, Math.round((totalHours / goal) * 100));
    
    var html = "<div style='font-family:sans-serif;max-width:600px;margin:20px auto;border:1px solid #e2e8f0;border-radius:12px;overflow:hidden;box-shadow:0 4px 6px -1px rgba(0,0,0,0.1);'>" +
      "<div style='background:#1e40af;padding:30px;color:white;text-align:center;'>" +
        "<h1 style='margin:0;font-size:24px;text-transform:uppercase;letter-spacing:1px;'>" + termName + " Impact Summary</h1>" +
        "<div style='font-size:14px;opacity:0.8;margin-top:5px;'>" + _toDMY_(start) + " — " + _toDMY_(end) + "</div>" +
      "</div>" +
      "<div style='padding:30px;background:white;'>" +
        "<div style='display:flex;justify-content:space-around;text-align:center;margin-bottom:30px;'>" +
          "<div style='flex:1;'>" +
            "<div style='font-size:32px;font-weight:bold;color:#1e40af;'>" + totalHours.toFixed(1) + "</div>" +
            "<div style='font-size:12px;color:#64748b;text-transform:uppercase;'>Total Hours</div>" +
          "</div>" +
          "<div style='flex:1;border-left:1px solid #e2e8f0;border-right:1px solid #e2e8f0;'>" +
            "<div style='font-size:32px;font-weight:bold;color:#1e40af;'>" + progress + "%</div>" +
            "<div style='font-size:12px;color:#64748b;text-transform:uppercase;'>Goal Reached</div>" +
          "</div>" +
          "<div style='flex:1;'>" +
            "<div style='font-size:32px;font-weight:bold;color:#1e40af;'>" + highImpactCount + "</div>" +
            "<div style='font-size:12px;color:#64748b;text-transform:uppercase;'>High Impact</div>" +
          "</div>" +
        "</div>" +
        
        "<h3 style='border-bottom:2px solid #f1f5f9;padding-bottom:10px;color:#334155;margin-top:40px;'>Engagement by Type</h3>" +
        "<div style='margin-top:15px;'>";
        
    Object.keys(relTypes).sort().forEach(function(key) {
        var percent = Math.round((relTypes[key] / totalHours) * 100);
        html += "<div style='margin-bottom:12px;'>" +
          "<div style='display:flex;justify-content:space-between;font-size:13px;margin-bottom:4px;'>" +
            "<span>" + key + "</span>" +
            "<span style='font-weight:bold;'>" + relTypes[key].toFixed(1) + "h</span>" +
          "</div>" +
          "<div style='height:8px;background:#f1f5f9;border-radius:4px;overflow:hidden;'>" +
            "<div style='height:100%;width:" + percent + "%;background:#3b82f6;'></div>" +
          "</div>" +
        "</div>";
    });
    
    html += "</div>" +
      "<h3 style='border-bottom:2px solid #f1f5f9;padding-bottom:10px;color:#334155;margin-top:40px;'>KPI Achievements</h3>" +
      "<ul style='list-style:none;padding:0;margin-top:15px;'>";
      
    Object.keys(kpis).sort().forEach(function(key) {
        html += "<li style='padding:8px 0;border-bottom:1px dashed #f1f5f9;font-size:13px;display:flex;justify-content:space-between;'>" +
          "<span>" + key + "</span>" +
          "<span style='background:#eff6ff;color:#1e40af;padding:2px 8px;border-radius:12px;font-weight:bold;'>" + kpis[key] + "</span>" +
        "</li>";
    });
    
    html += "</ul>" +
      "<div style='margin-top:40px;padding:20px;background:#f8fafc;border-radius:8px;font-size:13px;color:#64748b;line-height:1.6;'>" +
        "<strong>Note:</strong> This report is an automated summary of your logged interactions. High impact interactions are defined as those with an impact rating of 4 or 5." +
      "</div>" +
      "</div>" +
      "<div style='background:#f1f5f9;padding:20px;text-align:center;font-size:11px;color:#94a3b8;'>" +
        "Sent via Youth Work Tracker • " + new Date().toLocaleDateString() +
      "</div>" +
      "</div>";
      
    GmailApp.sendEmail(recipientEmail, "📊 Term Summary: " + termName, "", { htmlBody: html });
    return { ok: true, message: "Summary sent to " + recipientEmail };
  } catch (e) {
    return { ok: false, error: "Failed to send summary: " + e };
  }
}
